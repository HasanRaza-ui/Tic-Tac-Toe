void CComputerPlayer::makeMove(CBoard& board) {
	if (m_computerPlayer1) {
		if (m_countMove == 0) {
			board.placeStone(2, 2, m_symbol);
			m_countMove++;
		} else if (m_countMove == 2) {
			if (!board.checkPosition(1, 1)) {
				if (!board.checkPosition(2, 1) && !board.checkPosition(2, 0)) {
					board.placeStone(2, 0, m_symbol);
					m_countMove++;
				} else if (!board.checkPosition(1, 2) && !board.checkPosition(0, 2)) {
					board.placeStone(0, 2, m_symbol);
					m_countMove++;
				}
			}
		} else if (m_countMove == 4) {
			// Block opponent if they have two in a row, otherwise make a strategic move
			if (board.checkPosition(0, 0) && board.checkPosition(2, 2) && board.checkPosition(1, 1)) {
				board.placeStone(0, 2, m_symbol);
			} else if (board.checkPosition(0, 2) && board.checkPosition(2, 0) && board.checkPosition(1, 1)) {
				board.placeStone(0, 0, m_symbol);
			} else {
				// Make a strategic move
				if (!board.checkPosition(0, 1)) {
					board.placeStone(0, 1, m_symbol);
				} else if (!board.checkPosition(1, 0)) {
					board.placeStone(1, 0, m_symbol);
				} else if (!board.checkPosition(1, 2)) {
					board.placeStone(1, 2, m_symbol);
				} else if (!board.checkPosition(2, 1)) {
					board.placeStone(2, 1, m_symbol);
				}
			}
			m_countMove++;
		} else if (m_countMove == 6) {
			// Check for a winning move or block the opponent
			if (board.checkForWinningMove(m_symbol)) {
				// Make the winning move
				board.placeStone(board.getWinningMove().first, board.getWinningMove().second, m_symbol);
			} else if (board.checkForWinningMove(board.getOpponentSymbol())) {
				// Block the opponent's winning move
				board.placeStone(board.getWinningMove().first, board.getWinningMove().second, m_symbol);
			} else {
				// Make a strategic move
				// You can add more logic here
			}
			m_countMove++;
		} else if (m_countMove == 8) {
			// Check for a winning move or block the opponent
			if (board.checkForWinningMove(m_symbol)) {
				// Make the winning move
				board.placeStone(board.getWinningMove().first, board.getWinningMove().second, m_symbol);
			} else if (board.checkForWinningMove(board.getOpponentSymbol())) {
				// Block the opponent's winning move
				board.placeStone(board.getWinningMove().first, board.getWinningMove().second, m_symbol);
			} else {
				// Make a strategic move
				// You can add more logic here
			}
			m_countMove++;
